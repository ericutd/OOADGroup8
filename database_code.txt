create table administrator (
    userId int,
    name varchar(250),
    password varchar(250),
    email varchar(250)
    );
    
create table user (
    userId int,
    name varchar(250),
    password varchar(250),
    email varchar(250),
    accountType varchar(250),
    
    primary key(userId)
    );

create table permit (
    permitId int,
    ownerId int,
    price double,
    colorClass varchar(250),
    expirationDate datetime,
    
    primary key(permitId),
    foreign key(ownerId) references user(userId)
    );
    
create table vehicle (
    ownerId int,
    permitId int,
    licenseNum varchar(10),
    make varchar(10),
    model varchar(10),
    year varchar(10),
    color varchar(10),
    
    primary key(licenseNum),
    foreign key(ownerId) references user(userId),
    foreign key(permitId) references permit(permitId)
    );
    
create table parkingLot (
    parkingLotId int,
    
    primary key(parkingLotId)
    );

create table parkingSpot (
    parkingSpotId int,
    parkingLotId int,
    currentVehicle varchar(10),
    occupied boolean,
    colorClass varchar(250),
    
    primary key(parkingSpotId),
    foreign key(parkingLotId) references parkingLot(parkingLotId),
    foreign key(currentVehicle) references vehicle(licenseNum)
    );
